017package org.spongepowered.configurate.examples;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.ScopedConfigurationNode;
021import org.spongepowered.configurate.hocon.HoconConfigurationLoader;
022import org.spongepowered.configurate.transformation.ConfigurationTransformation;
023
024import java.io.IOException;
025import java.nio.file.Paths;
026
027import static org.spongepowered.configurate.transformation.NodePath.path;
028
029/**
030 * An example of how to use transformations to migrate a configuration to a newer schema version.
031 * <p>
032 * It's like DFU but not hot garbage! (and probably less PhD-worthy)
033 */
034public class Transformations {
035    private static final int VERSION_LATEST = 1; // easy way to track the latest version, update as more revisions are added
036    private static final String VERSION_KEY = "version";
037
038    /**
039     * Create a new builder for versioned configurations. This builder uses a field in the node (by default {@code
040     * schema-version}) to determine the current schema version (using -1 for no version present).
041     *
042     * @param <N> node type
043     * @return versioned transformation
044     */
045    public static <N extends ScopedConfigurationNode<N>> ConfigurationTransformation<N> create() {
046        return ConfigurationTransformation.<N>versionedBuilder()
047                .setVersionKey(VERSION_KEY) // this is the default, but to show it can be customized
048                .addVersion(VERSION_LATEST, zeroToOne()) // syntax: target version, latest version
049                .addVersion(0, initialTransform())
050                .build();
051    }
052
053    /**
054     * A transformation. This one has multiple actions, and demonstrates how wildcards work
055     *
056     * @param <N> node type
057     * @return created transformation
058     */
059    public static <N extends ScopedConfigurationNode<N>> ConfigurationTransformation<N> initialTransform() {
060        return ConfigurationTransformation.<N>builder()
061                // Move the node at `serverVersion` to the location <code>{"server", "version"}</code>
062                .addAction(path("serverVersion"), (path, value) -> {
063                    return new Object[]{"server", "version"};
064                })
065                // For every direct child of the `section` node, set the value of its child `new-value` to something
066                .addAction(path("section", ConfigurationTransformation.WILDCARD_OBJECT), (path, value) -> {
067                    value.getNode("new-value").setValue("i'm a default");
068
069                    return null; // don't move the value
070                })
071                .build();
072    }
073
074    public static <N extends ScopedConfigurationNode<N>> ConfigurationTransformation<N> zeroToOne() {
075        return ConfigurationTransformation.<N>builder()
076                // oh, turns out we want to use a different format for this, so we'll change it again
077                .addAction(path("server", "version"), (path, value) -> {
078                    @Nullable String val = value.getString();
079                    if (val != null) {
080                        value.setValue(val.replaceAll("-", "_"));
081                    }
082                    return null;
083                })
084                .build();
085    }
086
087    /**
088     * Apply the transformations to a node
089     * <p>
090     * This method also prints information about the version update that occurred
091     *
092     * @param node The node to transform
093     * @param <N>  node type
094     * @return provided node, after transformation
095     */
096    public static <N extends ScopedConfigurationNode<N>> N updateNode(N node) {
097        if (!node.isVirtual()) { // we only want to migrate existing data
098            ConfigurationTransformation<N> trans = create();
099            int startVersion = node.getNode(VERSION_KEY).getInt(-1);
100            trans.apply(node);
101            int endVersion = node.getNode(VERSION_KEY).getInt();
102            if (startVersion != endVersion) { // we might not have made any changes
103                System.out.println("Updated config schema from " + startVersion + " to " + endVersion);
104            }
105        }
106        return node;
107    }
108
109    public static void main(String[] args) throws IOException {
110        if (args.length != 1) {
111            System.err.println("Not enough arguments, usage: transformations <file>");
112            System.err.println("Apply the test transformations to a single file");
113        }
114        HoconConfigurationLoader loader = HoconConfigurationLoader.builder()
115                .setPath(Paths.get(args[0]))
116                .build();
117
118        loader.save(updateNode(loader.load())); // tada
119    }
120
121}
